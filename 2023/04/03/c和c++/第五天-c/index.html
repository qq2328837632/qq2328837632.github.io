多态

多态的基本概念

多态是C++面向对象三大特性之一

多态分为两类:

  • 静态多态:函数重载和运算符重载属于静态多态，复用函数名

    • ```
      #include

      // 函数重载
      void print(const int &num)
      {
      std::cout << “Printing integer: “ << num << std::endl;
      }

      void print(const float &num)
      {
      std::cout << “Printing float: “ << num << std::endl;
      }

      // 运算符重载
      class Vector2D
      {
      public:
      int x;
      int y;

      Vector2D(int xVal, int yVal) : x(xVal), y(yVal) {}

      // 类型转换运算符重载
      operator int()
      {
      return x + y;
      }
      };

      // 运算符重载
      Vector2D operator+(const Vector2D &v1, const Vector2D &v2)
      {
      return Vector2D(v1.x + v2.x, v1.y + v2.y);
      }

      int main()
      {
      int intValue = 10;
      float floatValue = 3.14;

      print(intValue); // 调用 print(const int&)
      print(floatValue); // 调用 print(const float&)

      Vector2D v1(1, 2);
      Vector2D v2(3, 4);
      Vector2D v3 = v1 + v2;
      int sum = v3; // 调用类型转换运算符重载

      std::cout << “Vector sum: “ << sum << std::endl;

      return 0;
      }

当子类重写父类的虚函数
这里不是子类中的虚函数表内部会替换成子类的虚函数地址，而是添加，因为替换的话，子类的虚函数就会丢失父类的虚函数地址，但实际上子类的虚函数表中同时有父类的虚函数地址和它自己的虚函数地址，通过使用方式不同，如利用父类指针或引用指向子类这种方式，来区分到底是使用子类虚数表里两个虚数表中的哪一个

// 基类模板
template
class Base {
public:
void func() {
// 调用派生类中的具体实现
static_cast<Derived*>(this)->implementation();
}
};

// 派生类
class Derived : public Base {
public:
void implementation() {
// 派生类中的具体实现
std::cout << “Derived::implementation()” << std::endl;
}
};

int main() {
Derived d;
d.func(); // 输出：Derived::implementation()

return 0;

}

class Base{
public:
//纯虚函数
//类中只要有一个纯虚函数就称为抽象类
//抽象类无法实例化对象,但是能定义一个指向该类的指针
//子类必须重写父类中的纯虚函数，否则也属于抽象类
virtual void func() = 0;
};
class Son :public Base{
public:
virtual void func(){
cout << “func调用”<< endl;
};
};
void teste1(){
Base * base = NULL;

class Animal {
public:
Animal(){
cout << “Animal构造函数调用!” << endl;
}
virtual void Speak() = 0;
//析构函数加上virtual关键字，变成虚析构函数
//virtual Animal()
//{
//cout << “Animal虚析构函数调用! “<<endl;
//}
virtual ~Animal() = 0;
};
Animal : :Animal(){
cout <<”Animal纯虚析构函数调用!” << endl;
}
//和包含普通纯虚函数的类一样，包含了纯虚析构函数的类也是一个抽象类。不能够被实例化。
class Cat : public Animal {
public:
Cat(string name)
{
cout << “cat构造函数调用!” <<endl;
m_Name = new string( name);
}
virtual void Speak(){
cout << *m_Name <<”小猫在说话!”<< endl;
}
~Cat()
{
cout << “cat析构函数调用!”<< endl;
if (this->m_Name != NULL){
delete m_Name;
m_Name = NULL;//置空是为了防止出现野指针
}
}
public:
string *m_Name;
};
void test01()
{
Animal *animal = new Cat( “Tom” ) ;
animal->Speak();
//通过父类指针去释放，会导致子类对象可能清理不干净，造成内存泄漏
//怎么解决?给基类增加一个虚析构函数
//虚析构函数就是用来解决通过父类指针释放子类对象
delete animal;
}
